Proceso SistemaReservacionesRestaurante
	// Definición de estructuras
	Dimension reservaciones[100]
	
	// Variables globales
	Definir maxCapacity, tablesAvailable Como Entero
	maxCapacity <- 50
	tablesAvailable <- 10
	Definir opcion Como Entero
	
	// Menú principal
	Repetir
		MostrarMenuPrincipal()
		Leer opcion
		Segun opcion Hacer
			Caso 1:
				RegistrarReservacion()
			Caso 2:
				MostrarReservaciones()
			Caso 3:
				CancelarReservacion()
			Caso 4:
				Escribir "Saliendo del sistema..."
			De Otro Modo:
				Escribir "Opción no válida, intente nuevamente"
		FinSegun
	Hasta Que opcion = 4
FinProceso

SubProceso MostrarMenuPrincipal()
	Escribir "===== SISTEMA DE RESERVACIONES ====="
	Escribir "1. Nueva Reservación"
	Escribir "2. Ver Reservaciones"
	Escribir "3. Cancelar Reservación"
	Escribir "4. Salir"
	Escribir "=================================="
	Escribir "Ingrese una opción: " Sin Saltar
FinSubProceso

SubProceso RegistrarReservacion()
	Definir nombre, contacto, fecha, hora Como Cadena
	Definir partySize, i Como Entero
	Definir confirmacion Como Caracter
	
	Escribir "===== NUEVA RESERVACIÓN ====="
	Escribir "Nombre del cliente: " Sin Saltar
	Leer nombre
	Escribir "Número de contacto: " Sin Saltar
	Leer contacto
	Escribir "Cantidad de personas: " Sin Saltar
	Leer partySize
	Escribir "Fecha (YYYY-MM-DD): " Sin Saltar
	Leer fecha
	Escribir "Hora (HH:MM): " Sin Saltar
	Leer hora
	
	// Validar disponibilidad
	Si partySize > 20 Entonces
		Escribir "Error: La cantidad de personas no puede superar 20."
		Retornar
	FinSi
	
	Si tablesAvailable <= 0 Entonces
		Escribir "No hay mesas disponibles en este momento."
		Retornar
	FinSi
	
	// Confirmación
	Escribir "Confirmar reservación (S/N): " Sin Saltar
	Leer confirmacion
	Si Mayusculas(confirmacion) = "S" Entonces
		Para i <- 1 Hasta 100 Hacer
			Si reservaciones[i] = "" Entonces
				reservaciones[i] <- Concatenar(nombre, " - ", contacto, " - ", fecha, " ", hora, " - ", partySize, " personas")
				tablesAvailable <- tablesAvailable - 1
				Escribir "Reservación registrada exitosamente."
				Salir
			FinSi
		FinPara
	FinSi
FinSubProceso

SubProceso MostrarReservaciones()
	Definir i Como Entero
	Escribir "===== RESERVACIONES EXISTENTES ====="
	Para i <- 1 Hasta 100 Hacer
		Si reservaciones[i] <> "" Entonces
			Escribir "Reservación ", i, ": ", reservaciones[i]
		FinSi
	FinPara
	Escribir "Presione Enter para continuar..."
	Esperar Tecla
FinSubProceso

SubProceso CancelarReservacion()
	Definir id Como Entero
	Escribir "Ingrese el número de la reservación a cancelar: " Sin Saltar
	Leer id
	Si id >= 1 Y id <= 100 Entonces
		Si reservaciones[id] <> "" Entonces
			reservaciones[id] <- ""
			tablesAvailable <- tablesAvailable + 1
			Escribir "Reservación cancelada exitosamente."
		Sino
			Escribir "No se encontró la reservación."
		FinSi
	Sino
		Escribir "ID inválido."
	FinSi
	Escribir "Presione Enter para continuar..."
	Esperar Tecla
FinSubProceso
